/** * 수정된 AI 분석 함수 * 1단계: 검색 도구를 사용해 유튜브 영상 정보를 텍스트로 가져옴 * 2단계: 가져온 텍스트를 AI에게 전달하여 구조화 */ async function analyzeSermon(youtubeUrl) { const apiKey = localStorage.getItem("gemini_api_key"); const API_URL = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`; // 1단계: 정보 수집 (JSON 응답 설정 제거 - 오류 방지) const response1 = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `다음 유튜브 설교 영상을 분석하여 제목, 본문, 요약, 핵심 대지, 예화를 추출해줘: ${youtubeUrl}` }] }], tools: [{ google_search: {} }] }) }); const data1 = await response1.json(); const extractedText = data1.candidates[0].content.parts[0].text; // 2단계: 수집된 텍스트를 구조화 (중요: responseMimeType 설정 제외) const response2 = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: `이 텍스트를 기반으로 JSON 형식의 데이터로 변환해줘. 필드: title, scripture, summary, points(배열), illustrations(배열). 텍스트: ${extractedText}` }] }] }) }); return await response2.json(); }
